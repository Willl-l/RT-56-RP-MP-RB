AUS_is_on_historical_plan_trigger = {
	OR = {
		AND = {
			is_historical_focus_on = yes
			has_game_rule = {
				rule = AUS_ai_behavior
				option = DEFAULT
			}
		}
		has_country_flag = AUS_AI_RANDOM_AUTHORITARIAN_HISTORICAL
		has_game_rule = {
			rule = AUS_ai_behavior
			option = AUTHORITARIAN_HISTORICAL
		}
	}
}
