GUA_fascist_boy_perfers_GER_over_JAP = {
	enable = {
		original_tag = GUA
		GUA = {
			has_government = fascism
		}
		country_exists = GUA
		country_exists = GER
		GER = {
			has_government = fascism
			is_faction_leader = yes
			surrender_progress < 0.05
		}
		JAP = {
			has_government = fascism
			is_faction_leader = yes
		}
	}
	
	abort_when_not_enabled = yes
	
	ai_strategy = {
		type = befriend
		id = "GER"	
		value = 25
	}
	
	ai_strategy = {
		type = befriend
		id = "JAP"	
		value = -75
	}

}

	enable = {
		original_tag = GUA
		GUA = {
			AND = {
				exists = yes
				has_government = fascism
				OR = {
					is_in_faction_with = ITA
					is_in_faction_with = GER
					is_in_faction_with = JAP
				}
				OR = {
				 	has_war_with = USA
					has_war_with = ENG
				}

		value = 100
	}

}
