add_namespace = FIN_elections



# Election of 1937: Svinhufvund vs Kallio
country_event = {
	id = FIN_elections.1
	title = FIN_elections.1.t
	desc = FIN_elections.1.d
	picture = GFX_report_event_election_vote

	fire_only_once = yes

	trigger = {
		tag = FIN
		date > 1937.1.1
		date < 1937.1.21
		OR = {
			has_government = democratic	
			has_government = neutrality
		}
		
	}
	
	option = { 	
		name = FIN_elections.1.a
		ai_chance = {
			factor = 25
			modifier = {
				factor = -25
				is_historical_focus_on = yes
			}
		}		
		set_party_name = { ideology = democratic long_name = FIN_kokoomusparty_long name = FIN_kokoomusparty }
		add_popularity = {
			ideology = democratic
			popularity = 0.1
		}
	}

	option = { 
		name = FIN_elections.1.b
		ai_chance = {
			factor = 25
			modifier = {
				factor = 100
				is_historical_focus_on = yes
			}
		}
			set_politics = {
			ruling_party = neutrality
			elections_allowed = yes
		}
		add_popularity = {
			ideology = neutrality
			popularity = 0.05
		}
		
	}
}

# 1940 Kyösti Kallio resigns
country_event = {
	id = FIN_elections.2
	title = FIN_elections.2.t
	desc = FIN_elections.2.d
	picture = GFX_report_event_election_vote

	fire_only_once = yes

	trigger = {
		tag = FIN
		date > 1940.12.1
		date < 1940.12.21
		has_government = neutrality
		NOT = {
			has_cosmetic_tag = SCA
		}
	}
	
	option = { 	
		name = FIN_elections.2.a
		kill_country_leader = yes
		set_party_name = { ideology = democratic long_name = FIN_edistyspuolue_long name = FIN_edistyspuolue }
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
		add_popularity = {
			ideology = democratic
			popularity = 0.1
		}
		create_country_leader = {
		name = "Risto Heikki Ryti"
		desc = "POLITICS_RISTO_HEIKKI_RYT_DESC"
		picture = "gfx/leaders/FIN/Portrait_Finland_Risto_Heikki_Ryti.dds"
		expire = "1965.1.1"
		ideology = liberalism
		traits = {
			}
		}
	}
}
