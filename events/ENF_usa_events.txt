	id = usamod.3
	title = usamod.3.t
	desc = usamod.3.d
	picture = GFX_report_event_chamberlain_announce
	
	is_triggered_only = yes
	
	option = { # Push Ahead
		name = usamod.3.a
		ai_chance = { factor = 80 }
		USA = {
			declare_war_on = {
				target = CAN
				type = puppet_wargoal_focus
			}
		}
	}
	option = { # Back Down
		name = usamod.3.b
		ai_chance = { factor = 20 }
		USA = {
			add_stability = -0.1
			add_political_power = -100
		}
	}
}
