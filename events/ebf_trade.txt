country_event = {
	id = ebf_trade.1
	title = ebf_trade.1.t
	desc = ebf_trade.1.d
	picture = GFX_report_event_german_speech
	
	is_triggered_only = yes
	
 	immediate = {
		while_loop_effect = {
			limit = {
				OR = {
					ROOT = {
						check_variable = {
							civ_amount_fix1 > 0
						}
					}
					ROOT = {
						check_variable = {
							mil_amount_fix1 > 0
						} 
					}
					ROOT = {
						check_variable = {
							dock_amount_fix1 > 0
						} 
					}
				}
			}
			if = {
				limit = { 
					ROOT = {
						check_variable = {
							civ_amount_fix1 > 0
						}  
					}
				}
				ROOT = {
					random_owned_controlled_state = {
						limit = {
							free_building_slots = {
								building = industrial_complex
								size > 0
								include_locked = no
							}
						}
						add_building_construction = {
							type = industrial_complex
							level = 1
						}
					}
					subtract_from_variable = { civ_amount_fix1 = 1 } 
				}
			}

			if = {
				limit = { 
					ROOT = {
						check_variable = {
							mil_amount_fix1 > 0
						}  
					}
				}
				ROOT = {
					random_owned_controlled_state = {
						limit = {
							free_building_slots = {
								building = arms_factory
								size > 0
								include_locked = no
							}
						}
						add_building_construction = {
							type = arms_factory
							level = 1
						}
					}
					subtract_from_variable = { mil_amount_fix1 = 1 } 
				}
			}

			if = {
				limit = { 
					ROOT = {
						check_variable = {
							dock_amount_fix1 > 0
						}  
					}
				}
				ROOT = {
					random_owned_controlled_state = {
						limit = {
							free_building_slots = {
								building = dockyard
								size > 0
								include_locked = no
							}
						}
						add_building_construction = {
							type = dockyard
							level = 1
						}
					}
					subtract_from_variable = { dock_amount_fix1 = 1 } 
				}
			}
		}

		ROOT = {
			set_variable = { 
				trade1 = 0
			}
			subtract_from_variable = {
				trades = 1
			}
		}
 	}
	option = {
		name = ebf_trade.1.a
	}
}
